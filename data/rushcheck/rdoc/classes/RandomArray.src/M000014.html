  def self.create_array(len)
    RushCheck::Gen.new do |n, r|
      ary = [self.create_components(@@base, n, r)]
      r2 = r
      (1..len).each do |i|
        r1, r2 = r2.split
        ary << self.create_components(@@indp.call(ary, i), n, r1)
      end
      
      ary
    end
  end
