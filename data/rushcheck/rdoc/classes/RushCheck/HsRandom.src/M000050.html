    def random_array(gen, len=nil, lo=nil, hi=nil)
      g = gen
      if len.nil?
      then
        Proc.new do |i|
          (i+1).times do
            v, g = random(g, lo, hi)
          end
          v
        end
      else
        (1..len).map do
          v, g = random(g, lo, hi)
          v
        end
      end
    end
